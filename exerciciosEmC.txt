void importar(){
	char linha[45], nomeAux[30], salAux[7], matAux[5];
	int i, j, k, l;	
	
	arquivo = fopen(ARQUIVO, "r");
	fgets(linha, 50, arquivo);
	while(!feof(arquivo)){
		for (i=0;i<30; i++){
			nomeAux[i] = linha[i];
		}
		j = 0;
		for (i=31;i<=36;i++){
			matAux[j] = linha[i];
			j++;
		}
		k=0;
		for (i=37;i<44;i++){
			salAux[k] = linha[i];
			k++;
		}
		printf("%s %s %s \n", nomeAux, matAux, salAux);
		fgets(linha, 50, arquivo);
		strcpy(pessoa.nome, nomeAux);
		pessoa.matricula = atoi(matAux);
		pessoa.salario = atof(salAux);
		
		arqSaida=fopen(ARQUIVOSAIDA, "w");
		for(l=0; l<=10; l++) {
			fwrite(&pessoa, sizeof(Pessoal), 1, arqSaida);
		}
	}
	fclose(arquivo);
	fclose(arqSaida);
}

	setlocale(LC_ALL, "Portuguese");
	
	char frase[] = "parâmetro por referência";
	
	printf("Frase: %s.\n", frase);
	inverteFrase(frase);
	printf("Após a inversão: %s", frase);
}

char inverteFrase(char *frase){
	int tamanho = 0;
	tamanho = strlen(frase);
	
	int c, i, j;

    for (i = 0, j = tamanho - 1; i < j; i++, j--)
    {
        c = frase[i];
        frase[i] = frase[j];
        frase[j] = c;
    }
    return frase;
}
